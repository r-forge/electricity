\subsection{Maximierungsproblem eines Spielers}

\begin{frame}{Maximierungsproblem eines Spielers}

\begin{gather}
	\max \pi_i(q_{i}^{s,t},K_{i}^t,I_{i}^t)=
	\sum_{s\in S} \rho_s \left[ ((\alpha^s- \beta Q^{s,1}) - c) q_{i}^{s,1}  \right] - \Gamma I_{i}^{0} \label{eq:oligopmax2} \\
			\text{s.t.:} \  q_{i}^{s,1} - K_{i}^t \leq 0; \ \forall i,s \label{eq:capacitycon} \\ 
										  K^{1}_{i}  - K^{0}_{i}  - I_{i}^0 = 0 ; \ \forall i  \label{eq:state} \\
										   q_{i}^{s,1}; K^t_{i}; I_{i}^0	\geq 0; \ \forall i,s,t  \nonumber
\end{gather}

{\small
\begin{tabbing}
wobei: \= $\rho_s$ \= Wahrscheinlichkeit für Szenario /  Market State\\
\> $I_{i}^{0}$     \> Investitionen - action variable  \\
\> $K_{i}^t $      \>  Kapazität - Statusvariable \\
\> $\Gamma$        \> Investitionskosten \\
\> $c$             \> Grenzkosten \\
\end{tabbing}}
\end{frame}

\begin{frame}{Maximierungsproblem eines Spielers II}

Die dazugehörige Hamiltonian in diskreter Zeit:

\begin{gather}
	 H_i(q_{i}^{s,t},K_{i}^t,I_{i}^t,\lambda_{i}^{s,1},u_{i}^1)= 
	 \sum_{s\in S} \rho_s \left[((\alpha^s- \beta Q^{s,1}) - c) q_{i}^{s,1} \right]	- \Gamma I_{i}^{0}  \\ \nonumber  
	  	- \lambda_{i}^{s,1}(q_{i}^{s,1} - K^{1}_{i}) \\ \nonumber
			- u_{i}^1(K^{1}_{i}  - K^{0}_{i}  - I_{i}^t)	\\  \nonumber
\end{gather}

{\small
\begin{tabbing}
wobei: \= $\lambda_{i}^{s,1}$ \= Schattenpreis einer Einheit Kapazität für Spieler $i$ in Szenario $s$\\
\> $u_{i}^1$     \>  Schlupfvariable für die Bewegungsgleichung\\
\end{tabbing}}

\end{frame}

\subsection{Die KKT Bedingungen}

\begin{frame}{KKT Bedingungen I}


\begin{gather}
\frac{\partial H}{ \partial q_{i}^{s,1}}: - \rho_s \left[ \alpha^s - \beta q_{i}^{s,1} - \beta Q^{s,1} - c \right] + \lambda_{i}^{s,1} \geq 0 \ \bot \ q_{i}^{s,1} \geq 0;\ \forall i,s \label{eq:foc1a}
\end{gather}

Da Mengen immer strikt positiv, kann man schreiben

\begin{gather}
 \rho_s \left[ \alpha^s - \beta q_{i}^{s,1} - \beta Q^{s,1}\right]=  \rho_s c  + \lambda_{i}^{s,1} ;\ \forall i,s    \label{eq:foc1b}
\end{gather}

\begin{itemize}
	\item erwarteter Grenzertrag = marginale Kosten + Engpassrente
\end{itemize}

\begin{tabular}{ccc}
 \hline
perfekte Konkurrenz & Oligopol &  Monopol \\
$\alpha^s - \beta Q^{s,1}$ &  $\alpha^s - \beta q_{i}^{s,1} - \beta Q^{s,1}$    & $\alpha^s - 2* \beta Q^{s,1}$  \\
 \hline
\end{tabular}

\begin{itemize}
	\item imperfekte Konkurrenz verzerrt die Mengen nach unten
\end{itemize}

\end{frame}

\begin{frame}{KKT Bedingungen II}

\begin{gather}
\tfrac{\partial H}{ \partial K^1_{i}}:  -\sum_{s\in S}  \lambda_{i}^{s} + u_{i}^1 \geq 0 \ \bot \ K^1_{i} \geq 0 ; \forall i,k \label{eq:foc3} \\
\tfrac{\partial H}{ \partial I_{i}^{0}}: \Gamma - u_{i}^1 \geq 0 \ \bot \ I_{i}^{0} \geq 0 ;\forall i \label{eq:foc4}
\end{gather}

\begin{itemize}
	\item da $K^1_{i} > 0$, zeigt $u_{i}^1$ den zusätzlichen Gewinn durch eine Einheit mehr Kapazität
	\item es wird so lange investiert, bis der \textbf{zusätzliche Gewinn} gleich den \textbf{Kosten} ist
	\item Realoptionsinterpretation - nur Investition, wenn erwartete Gewinne gleich Investitionskosten
\end{itemize}

\begin{itemize}
	\item Ableitungen nach  $\lambda_{i}^{s,1}$ und $u_{i}^1$ ergeben die Kapazitätsbeschränkung und die Bewegungsgleichung
\end{itemize}

\end{frame}

\begin{frame}{KKT Bedingungen III}

Mengen, und damit Kapazitäten werden erweitert, bis gilt:

\begin{gather}
\sum_{s\in S}\lambda_{i}^{s} = \Gamma \label{eq:invcondition};  \forall i
\end{gather}

\begin{itemize}
	\item gilt für jede Markform
	\item weitere Definition von Engpassrenten als im klassischen Peak Load Pricing - Unterschied zwischen $c$ und Grenzerträgen
	\item Grenzerträge fallen unter imperfekter Konkurrenz einfach schneller in $q$
	\item da Preise über den Grenzerträgen - hohe Preise nicht notwendigerweise ein Investitionsanreiz
	\item bindende Kapazitäten, also Grenzerträge $>$ $c$ können in mehreren Szenarien auftreten
\end{itemize}

\end{frame}

\section{Price Caps}

\subsection{Modellierung von Price Caps}

\begin{frame}{Modellierung von Price Caps I}

\begin{gather}
	\max \pi_i(q_{i}^{s,t},K_{i}^t,I_{i}^t, \overline{p})= \label{eq:pcmax1}
	\sum_{s\in S} \rho_s \left[ ( \min{(p_{s,1}^{o},\overline{p})} - c) q_{i}^{s,1}  \right] - \Gamma I_{i}^{0}  \\
			\text{s.t.:} \  p_{s,1}^{o} - \overline{p} \leq 0; \forall s,t \label{eq:pcconstraint}
\end{gather}

\begin{itemize}
	\item Oligopol- Monopolpreis $p_{s,1}^{o}= \alpha^s - \beta Q^{s,1}$ darf nicht über $\overline{p}$ steigen
	\item Gewinnfunktion: entweder Oligopol- Monopolpreis oder einen fixen Betrag $\overline{p}$ pro verkaufter Einheit (nicht differenzierbar)
	\item zusätzliche Annahme: $s \in h,l$ und $\alpha^h > \alpha^l$
\end{itemize}

\end{frame}

\begin{frame}{Modellierung von Price Caps II}

\begin{itemize}
	\item Stufe 1: Modell ohne Price Cap lösen und $p_{s,1}^{o}$ und den optimalen Preis $p_{s,1}^{c}$ berechnen
	\item Stufe 2: drei Möglichkeiten \\
	Fall 1: wenn der Price Cap nicht bindend ist - fertig


\end{itemize}

\begin{table}[h]
	\centering
\begin{tabular}{ccc}
\hline
        &  Fall 2  & Fall 3  \\
\hline
        & $c+\Gamma \leq p_{h,1}^{c}$ & $c+\Gamma > p_{h,1}^{c}$ \\
\hline
\multicolumn{ 1}{c}{\textbf{a: }$\overline{p} \geq p_{h,1}^{c}$} & optimale Kapazitäten möglich  & \multicolumn{ 1}{c}{nicht optimal} \\
\multicolumn{ 1}{c}{} & at $\overline{p} = p_{h,1}^{c}$ & \multicolumn{ 1}{c}{} \\
\hline
\multicolumn{ 1}{c}{\textbf{b:} $\overline{p} < p_{h,1}^{c}$} & \multicolumn{ 1}{c}{nicht optimal} & zweitbester PC \\

\multicolumn{ 1}{c}{} & \multicolumn{ 1}{c}{} & at $c+\tfrac{\Gamma}{(\rho_h+\rho_l)}$ \\
\hline
\end{tabular}  
\end{table}

\end{frame}

\begin{frame}{Fall 2a}

Im Intervall 2a gibt es keine Diskontinuitäten und wir können folgende Zielfunktion verwenden:

\begin{gather}
	\max \pi_i(q_{i}^{s,t},K_{i}^t,I_{i}^t, \overline{p})= \label{eq:pcmax2}
	\sum_{s\in S} \rho_s \left[ (p_{s,1}^{o} - c) q_{i}^{s,1}  \right] - \Gamma I_{i}^{0}  \\  \nonumber 
\end{gather}

Bedingung erster Ordnung $\frac{\partial H}{ \partial q_{i}^{s,1}}$ änder sich:

\begin{gather}
 \rho_s \left[ \alpha^s - \beta q_{i}^{s,1} - \beta Q^{s,1}\right] + \beta \psi_s =  \rho_s c  + \lambda_{i}^{s,1}   ;\ \forall i,s  \label{eq:foc1c}
\end{gather}

\begin{itemize}
	\item wobei $\psi_s$ die Schlupfvariable für den PC darstellt
	\item je niedriger der PC, je mehr werden die Mengen und damit die Investitionen nach oben verzerrt
	\item könnte so gesetzt werden, dass die linke Seite dem Resultat under perfekter Konkurrenz entspricht
	\item aber Price Caps sind ein grobes Instrument
\end{itemize}
\end{frame}

\begin{frame}{Fall 2b und 3}

Sobald $\overline{p}$ niedriger als $p_{s,1}^{c}$ ist, Fixkosten in Spitzenperioden nicht gedeckt - Missing Money Problem

\begin{gather}
	\max \pi_i(q_{i}^{s,t},K_{i}^t,I_{i}^t, \overline{p})= \label{eq:pcmax3}
	 \rho_h \left[ (\overline{p}-c) q_{i}^{l,1}  \right]+ \rho_h \left[ (p_{l,1}^{o} - c) q_{i}^{l,1}  \right] - \Gamma I_{i}^{0}  \\  \nonumber
\end{gather}

$\overline{p}-c$ nur noch wie Subvention bei Entscheidung wieviel Kapazität für die Schwachlastperiode gebaut werden soll

\end{frame}

\section{Conclusio}


\begin{frame} {Conclusio}					


\begin{itemize}
	\item optimale Verhalten von Unternehmen in einem zweistufigen Investitionsspiel, unter Unsicherheit dargestellt
	\item mit Monopol- Oligopol- und Optimalfall verglichen
	\item Grund zur Sorge - durch strategisches Verhalten entsteht ein Anreiz, Kapazitäten zurückzufahren
\end{itemize}

\begin{itemize}
	\item Price Caps können die Investitionen sogar steigern
	\item in gewissen Fällen kann man sogar ein optimales Investitionsniveau erreichen
	\item bergen aber einige Risken, sobald das "`Missing Money Problem"' auftritt
	\item hoher Informationsbedarf
\end{itemize}

\end{frame}
