\section{Motivation und Literatur}		
		\begin{frame}{Motivation und Literatur}
		
		 \begin{itemize}
		 \item Kann man auf den momentanen Elektizitätsmärkten effiziente Investitionen erwarten?
		 
		 \begin{itemize}
		 \item Was bedeutet ausreichend? $\rightarrow$ Peak Load Pricing (Crew and Kleindorfer, 1986)
		 \item Wie wirkt ein Oligopol auf Investitionsentscheidungen? (von der Fehr and Harbord, 1995)
		 \item Wird möglicherweise auch die Technologiewahl verzerrt?
		 \end{itemize}

		 \item Abbildung der tatsächlichen Investitionsentscheidung in einem mehrstufigen Spiel
		 
\begin{itemize}
\item Annahmen über Informationen die die Spieler berücksichtigen (Cellini und Lambertini, 2004)
\begin{itemize}
	\item Open loop $\rightarrow$ Anfangszustand des Spiels und Zeit
	\item S-adapted Open loop $\rightarrow$ zusätzlich Realisation der Zufallsvariablen
	\item Closed loop $\rightarrow$ zusätzlich Veränderungen der Statusvariablen
\end{itemize}
\item \emph{Game with Probabilistic Scenarios (GPS)} nach Genc (2007)
\end{itemize}
\item Wie weit weicht die tatsächliche von der optimalen Investitionsentscheidung ab?

\end{itemize}
\end{frame}

\begin{column} {0.5\textwidth}
\begin{itemize}
\item An der optimalen Kapazität ist der marginale Nutzen gleich den marginalen Kosten der Kapazität.
\end{itemize}

\begin{equation}
	c=(p^p-v)\times\pi
\end{equation}

{\small
\begin{tabbing}
mit: \= $c$ \  \= Kosten einer Einheit Kapazität \\
\> $p^p$   \    \> Engpasspreis  \\
\> $v$    \   \> variable Kosten \\
\> $\pi$    \    \> Wahrscheinlichkeit für Peak-Periode
\end{tabbing}}

\begin{itemize}
\item Genau das würde auch eine Firma unter vollkommener Konkurrenz tun.
\end {itemize}

\end{column}
\end{columns}

\begin{frame}

\frametitle{Verzerrte Investitionsanreize im Oligopol I}
\begin{columns}
\begin{column} {0.6\textwidth}

\begin{figure}[h]
\centering
\includegraphics[width=1.\textwidth]{imperfect_spot_pricing}
    \caption{von der Fehr and Harbord (1995)}
    \label{fig:Daten 2004}            
\end{figure}
\end{column}

\begin{column} {0.4\textwidth}
\begin{itemize}
\item Preise $>$ als die marginalen Kosten
\item zusätzlicher Investitionsanreiz durch Gewinne in Off-Peak Perioden
\end {itemize}
  
\begin{equation}
	1/2 (1-\pi) (p^{op}-v)
\end{equation} 
{\small
\begin{tabbing}
mit: \= $p^{op}$ \  \= Off-Peak Preis
\end{tabbing}}

\begin{itemize}
\item Optimum gleich wie vorher
\item Verzerrung in Richtung zu viel Investitionen
\end {itemize}

\end{column}
\end{columns}
	

\end{frame}
\begin{frame}

\frametitle{Verzerrte Investitionsanreize im Oligopol II}
\begin{columns}
\begin{column} {0.6\textwidth}

\begin{figure}[h]
\centering
\includegraphics[width=1.\textwidth]{imperfect_spot_pricing2}
    \caption{von der Fehr and Harbord (1994)}
   % \label{fig:Daten 2004}            
\end{figure}
\end{column}

\begin{column} {0.4\textwidth}
\begin{itemize}
\item wenn man weniger investiert, erhöht sich der Preis in Peak-Perioden
\end {itemize}

\begin{equation}
\pi (w^p(K)+\frac{\partial w^p(K)}{\partial q}-v)
\end{equation}

{\small
\begin{tabbing}
\= $w^p$ \  \= Nachfragefunktion in Peak Perioden \\
\> $K$   \    \> Kapazität
\end{tabbing}}

\begin{itemize}
\item die Veränderung des Preises durch mehr Kapazität wird nun im Oligopol berücksichtigt
\item dieser Effekt ist negativ für das Unternehmen, also Verzerrung Richtung zu wenig Investitionen
\end {itemize}

\end{column}
\end{columns}
\end{frame}

\section{Modell}

		\begin{frame}{Modell}
\begin{gather}
	\max \pi_i(q_{i,k}^{s,m,t},K^t_{i,k},I_{i,k})=
	\sum_{m\in M} v_m \left[ (\alpha_m^l-\beta_m^l \sum_{i\in N}\sum_{k\in K} q_{i,k}^{l,m,1}) \sum_{k\in K} q_{i,k}^{l,m,1} - \sum_{k\in K} c_k q_{i,k}^{l,m,1} \right]  \\  \nonumber 
	+ \delta \sum_{s\in S} p_s \sum_{m\in M} v_m \times  \left[ (\alpha_m^s-\beta_m^s \sum_{i\in N}\sum_{k\in K} q_{i,k}^{s,m,2}) \sum_{k\in K} q_{i,k}^{s,m,2} - \sum_{k\in K} c_k q_{i,k}^{s,m,2}  \right]   \\  \nonumber 
									- \sum_{k\in K} \Gamma_{k\in K} I_{i,k} + \delta \sum_{k\in K} F_k I_{i,k}  \\       
			\text{s.t.:} \  q_{i,k}^{l,m,t} - K^{t}_{i,k} \leq 0; \ \forall i,k,m,t    \label{eq:oligopmax2}\\ 
										  K^{2}_{i,k}  - \rho K^{1}_{i,k}  - I_{i,k} = 0 ; \ \forall i,k  \label{eq:ologopmax5} \\  
\end{gather}
\small
\begin{tabular}[c]{l l l l}
$i\in N$        & Spieler, Firmen &$K_{i,k}^t$& verfügbare Kapazität\\
$s\in S$       	& Szenarien &$q_{i,k}^{s,m,t}$&Mengen\\
$m\in M$	& Marktszenarien &$I_{i,k}^t$& Investitionen\\
$k\in K$	& Technologien &$p_s$ & Wahrscheinlichkeit für Szenario $s$\\
$t\in T$	& Zeit &$\delta$& Diskontfaktor\\
 
$\rho$   & Abschreibungsfaktor &$c_k$& variable Kosten\\
$\alpha_m$  & Interzept der Nachfragefunktion &$\Gamma_k$&Investitionskosten \\
$\beta_m$   & Anstieg der Nachfragefunktion  &$F_k$ & Restwert \\
\end{tabular}
\normalsize
		\end{frame}

\section{Conclusio}

\begin{frame}{Conclusio}
\begin{box}
\begin{itemize}
	\item Untersuchung von Investitionsanreizen auf E-Märkten ohne Berücksichtigung der strategischen Investitionen unmöglich
	\item rein analytisches Modell - zuwenig oder auch zuviel Investitionen
	\item realistisches Beispiel gerechnet
	\item Vergleich zwischen Wohlfahrtsoptimum und Oligopollösung zeigt ein Investitionsproblem
	\item durch Kernkraftausstieg und alternde Kapazitäten noch verschlimmert
\end{itemize}
\end{box}
